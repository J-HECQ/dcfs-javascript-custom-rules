异步JS：async/await 必须成对出现，否则运行时抛出异常。 

 
// Noncompliant 
function run1() {
	await waitFiles();
}

function run2(){
	waitFiles(function (){
		await doSomeThing();
	});
}

async function run3(){
	waitFiles(function (){
		await doSomeThing();
	});
}

请使用ColumnName获取行列索引，进行操作


// OK
async function run4() {
	await waitFiles();
}

function run5(){
	waitFiles(async function (){
		await doSomeThing();
	});
}
